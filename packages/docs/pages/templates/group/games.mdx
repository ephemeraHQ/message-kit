## Skill declaration

How the skill is declared in the `skills.ts` file:

```tsx [src/skills.ts]
export const skills: SkillGroup[] = [
  {
    name: "Group bot",
    tag: "@bot",
    description: "Group agent for tipping payments, games and more.",
    skills: [
      {
        skill: "/game [game]",
        handler: games,
        description: "Play a game.",
        params: {
          game: {
            default: "",
            type: "string",
            values: ["wordle", "slot", "help"],
          },
        },
      },
    ],
  },
];
```
