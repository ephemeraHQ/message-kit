# Tipping

In this example you can tip using 3 primitives:

```bash [cmd]
/tip @bo @alix 15
```

## Skill declaration

How the skill is declared in the `skills.ts` file:

```tsx [src/skills.ts]
export const skills: SkillGroup[] = [
  {
    name: "Group bot",
    tag: "@bot",
    description: "Group agent for tipping payments, games and more.",
    skills: [
      /* Other skills */
      {
        skill: "/tip [usernames] [amount] [token]",
        description: "Tip users in a specified token.",
        handler: tipping,
        params: {
          username: {
            plural: true,
            default: "",
            type: "username",
          },
          amount: {
            default: 10,
            type: "number",
          },
        },
      },
      /* Other skills */
    ],
  },
];
```

## Handler

The handler function to process tipping skill requests:

```tsx [src/handler/tipping.ts]
import {
  XMTPContext,
  AbstractedMember,
  SkillResponse,
} from "@xmtp/message-kit";
import { getUserInfo } from "@xmtp/message-kit";

export async function handler(context: XMTPContext): Promise<SkillResponse> {
  const {
    members,
    message: {
      content: {
        reference,
        reply,
        text,
        skill,
        params: { amount, username },
      },
      sender,
    },
  } = context;
  let receivers: AbstractedMember[] = [];

  receivers = await Promise.all(
    username.map((username: string) => getUserInfo(username)),
  );

  if (!sender || receivers.length === 0 || amount === 0) {
    context.reply("Sender or receiver or amount not found.");
    return {
      code: 400,
      message: "Sender or receiver or amount not found.",
    };
  }
  const receiverAddresses = receivers.map((receiver) => receiver.address);

  context.sendTo(
    `You received ${amount} tokens from ${sender.address}.`,
    receiverAddresses,
  );

  // Notify sender of the transaction details
  context.sendTo(
    `You sent ${amount * receiverAddresses.length} tokens in total.`,
    [sender.address],
  );
  return {
    code: 200,
    message: "Success",
  };
}
```
